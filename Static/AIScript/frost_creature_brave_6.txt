#name frost_creature_brave_6
#queue_jumps call

; 5057 - Frozen, 5 will, buff
; 5157 - Assault, 3 might, add 1 mcharge
; 5195 - Frost Spear, 4 will, add 1 wcharge
; 5217 - Jarnsaxa's Kiss - 3 will, need 1-5 wcharge
; 32766 - melee
; 5221 - Bash, 1 might, need 1-3 mcharge  (not found in ability scan)

set var 0
use 5057

:begin
call waitfortarget
use 32766

getwillcharge var
if var >= 2
  use 5217
  wait 1000
endif

getwill var
if var >= 4
  use 5195
  wait 2000
  goto begin
endif

getmightcharge var
if var >= 2
  use 5221
  wait 1000
endif

getmight var
if var >= 3
  use 5157
  wait 2000
endif 

goto begin

:waitfortarget
has_target var
if var = 0
  wait 2000
  goto waitfortarget
endif
return

:tryMelee
use 32766
return
