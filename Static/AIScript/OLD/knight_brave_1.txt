#name knight_brave_1

; 5157 - Assault, 3 might, add 1 mcharge
; 5221 - Bash, 1 might, need 1-3 mcharge
; 32766 - melee

set var 0

:begin
call waitfortarget
use 32766

getmightcharge var
if var >= 3
  use 5221
  wait 1000
endif

getmight var
if var >= 3
  use 5157
  wait 2000
endif

goto begin

:waitfortarget
has_target var
if var = 0
  wait 2000
  goto waitfortarget
endif
return
